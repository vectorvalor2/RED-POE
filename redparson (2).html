// ==================== THREE.JS ENGINE CORE ====================
let scene, camera, renderer;
let objects = [];
let audioContext;
let physicsEnabled = true;
function initEngine() {
const canvas = document.getElementById('engineCanvas');
renderer = new THREE.WebGLRenderer({ canvas, antialias: true });
renderer.setSize(1200, 520);
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.shadowMap.enabled = true;
scene = new THREE.Scene();
scene.background = new THREE.Color(0x0a0a0a);
camera = new THREE.PerspectiveCamera(60, 1200/520, 0.1, 1000);
camera.position.set(8, 6, 12);
camera.lookAt(0, 0, 0);
// Lighting - Golden ratio inspired
const ambient = new THREE.AmbientLight(0xaaaaaa, 0.6);
scene.add(ambient);
const dirLight = new THREE.DirectionalLight(0xffeecc, 1.1);
dirLight.position.set(10, 15, 8);
dirLight.castShadow = true;
scene.add(dirLight);
// Ground
const groundGeo = new THREE.PlaneGeometry(40, 40);
const groundMat = new THREE.MeshStandardMaterial({ color: 0x222222 });
const ground = new THREE.Mesh(groundGeo, groundMat);
ground.rotation.x = -Math.PI / 2;
ground.receiveShadow = true;
scene.add(ground);
// Initial objects
addTestObjects();
// Web Audio
audioContext = new (window.AudioContext || window.webkitAudioContext)();
animate();
}
function addTestObjects() {
// Central golden ratio object
const box = new THREE.Mesh(
new THREE.BoxGeometry(2, 2, 2),
new THREE.MeshStandardMaterial({ color: 0xff3366, metalness: 0.7 })
);
box.position.y = 1;
box.castShadow = true;
scene.add(box);
objects.push(box);
// Satellite objects
for (let i = 0; i < 6; i++) {
const sphere = new THREE.Mesh(
new THREE.SphereGeometry(0.6, 32, 32),
new THREE.MeshStandardMaterial({ color: 0x33ccff })
);
const angle = (i / 6) * Math.PI * 2;
sphere.position.set(
Math.cos(angle) * 5,
1.2,
Math.sin(angle) * 5
);
scene.add(sphere);
objects.push(sphere);
}
}
function animate() {
requestAnimationFrame(animate);
// Gentle golden ratio rotation on central object
if (objects.length > 0) {
objects[0].rotation.y += 0.003;
}
// Simple physics simulation
if (physicsEnabled) {
objects.forEach((obj, i) => {
if (i > 0) {
obj.position.y = 1.2 + Math.sin(Date.now() / 800 + i) * 0.3;
}
});
}
renderer.render(scene, camera);
}
function togglePhysics() {
physicsEnabled = !physicsEnabled;
alert(physicsEnabled ? "Physics Enabled" : "Physics Paused");
}

// Audio collision SFX
function playCollisionSFX(intensity) {
if (!audioContext) return;
const oscillator = audioContext.createOscillator();
const gain = audioContext.createGain();
oscillator.type = 'sawtooth';
oscillator.frequency.setValueAtTime(80 + intensity * 400, audioContext.currentTime);
gain.gain.value = Math.min(1, intensity * 0.8);
oscillator.connect(gain);
gain.connect(audioContext.destination);
oscillator.start();
gain.gain.exponentialRampToValueAtTime(0.001, audioContext.currentTime + 0.6);
oscillator.stop(audioContext.currentTime + 0.7);
}
